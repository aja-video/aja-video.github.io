480 CNTV2VPID & CNTV2VPID::SetColorimetry (const NTV2VPIDColorimetry inColorimetry)
481 {

483  if (standard == VPIDStandard_1080 ||
484  standard == VPIDStandard_1080_DualLink ||

486  standard == VPIDStandard_2160_QuadDualLink_3Gb ||
487  standard == VPIDStandard_2160_DualLink)

491  lowBit = inColorimetry&0x1;

495  ((lowBit << kRegShiftVPIDColorimetryAltLow) & kRegMaskVPIDColorimetryAltLow);
496  }
497  else
498  {
499  m_uVPID = (m_uVPID & ~kRegMaskVPIDColorimetry) |
500  ((ULWord(inColorimetry) << kRegShiftVPIDColorimetry) & kRegMaskVPIDColorimetry);
501  }
