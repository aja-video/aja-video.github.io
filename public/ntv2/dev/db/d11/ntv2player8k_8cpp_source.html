182 AJAStatus NTV2Player8K::SetUpVideo (void)
183 {
184  // Configure the device to output the requested video format...
185  if (mConfig.fVideoFormat == NTV2_FORMAT_UNKNOWN)
186  return AJA_STATUS_BAD_PARAM;
187  if (!mDevice.features().CanDoVideoFormat(mConfig.fVideoFormat))
188  { cerr << "## ERROR: '" << mDevice.GetDisplayName() << "' doesn't support "
189  << ::NTV2VideoFormatToString(mConfig.fVideoFormat) << endl;
190  return AJA_STATUS_UNSUPPORTED;
191  }
192  if (!mDevice.features().CanDoFrameBufferFormat(mConfig.fPixelFormat))
193  { cerr << "## ERROR: '" << mDevice.GetDisplayName() << "' doesn't support "
194  << ::NTV2FrameBufferFormatString(mConfig.fPixelFormat) << endl;
195  return AJA_STATUS_UNSUPPORTED;
196  }
197  if (::IsRGBFormat(mConfig.fPixelFormat) != mConfig.fDoRGBOnWire && mDevice.features().GetNumCSCs() == 0)
198  { cerr << "## ERROR: '" << mDevice.GetDisplayName() << "' has no CSCs, and '--pixelFormat' \""
199  << ::NTV2FrameBufferFormatToString(mConfig.fPixelFormat, true) << "\" contradicts '--rgb' setting" << endl;
200  return AJA_STATUS_UNSUPPORTED;
201  }
202 
203  NTV2ChannelSet channels13, frameStores;
204  channels13.insert(NTV2_CHANNEL1); channels13.insert(NTV2_CHANNEL3);
205  if (mConfig.fDoTsiRouting)
206  { // "Tsi" routing requires 2 FrameStores
207  if (channels13.find(mConfig.fOutputChannel) == channels13.end())
208  return AJA_STATUS_BAD_PARAM; // fOutputChannel not Ch1 or Ch3
209  frameStores = ::NTV2MakeChannelSet (mConfig.fOutputChannel, 2); // 2 FrameStores starting at fOutputChannel
210  }
211  else
212  { // "Squares" routing requires 4 FrameStores
213  if (mConfig.fOutputChannel != NTV2_CHANNEL1)
214  return AJA_STATUS_BAD_PARAM; // fOutputChannel not Ch1
215  frameStores = ::NTV2MakeChannelSet (mConfig.fOutputChannel, 4); // 4 FrameStores starting at fOutputChannel
216  }
217 
218  // Keep the raster description handy...
219  mFormatDesc = NTV2FormatDescriptor(mConfig.fVideoFormat, mConfig.fPixelFormat);
220  if (!mFormatDesc.IsValid())
221  return AJA_STATUS_FAIL;
222 
223  // Turn on the FrameStores (to read frame buffer memory and transmit video)...
224  mDevice.EnableChannels (frameStores, /*disableOthers=*/!mConfig.fDoMultiFormat);
225 
226  // This demo requires VANC be disabled...
227  mDevice.SetVANCMode (frameStores, NTV2_VANCMODE_OFF); // VANC is incompatible with 8K/UHD2 formats
228 
229  // Set the FrameStore video format...
230  mDevice.SetVideoFormat (mConfig.fVideoFormat, false, false, mConfig.fOutputChannel);
231  mDevice.SetQuadQuadFrameEnable (true, mConfig.fOutputChannel);
232  mDevice.SetQuadQuadSquaresEnable (!mConfig.fDoTsiRouting, mConfig.fOutputChannel);
233 
234  // Set the frame buffer pixel format for the device FrameStore(s)...
235  mDevice.SetFrameBufferFormat (frameStores, mConfig.fPixelFormat);
236 
237  // The output interrupt is Enabled by default, but on some platforms, you must subscribe to it
238  // in order to be able to wait on its event/semaphore...
239  mDevice.SubscribeOutputVerticalEvent (mConfig.fOutputChannel);
240 
241  // Get current per-field maximum Anc buffer size...
242  if (!mDevice.GetAncRegionOffsetFromBottom (gAncMaxSizeBytes, NTV2_AncRgn_Field2))
243  gAncMaxSizeBytes = NTV2_ANCSIZE_MAX;
244 
245  // Set output clock reference...
246  mDevice.SetReference(mDevice.features().CanDo2110() ? NTV2_REFERENCE_SFP1_PTP : NTV2_REFERENCE_FREERUN);
247 
248  // At this point, video setup is complete (except for widget signal routing).
249  return AJA_STATUS_SUCCESS;
250 
251 } // SetUpVideo

254 AJAStatus NTV2Player8K::SetUpAudio (void)
255 {
256  uint16_t numAudioChannels (mDevice.features().GetMaxAudioChannels());
257 
258  // If there are 8192 pixels on a line instead of 7680, reduce the number of audio channels
259  // This is because HANC is narrower, and has space for only 8 channels
260  if (NTV2_IS_UHD2_FULL_VIDEO_FORMAT(mConfig.fVideoFormat) && numAudioChannels > 8)
261  numAudioChannels = 8;
262 
263  // Use the NTV2AudioSystem that has the same ordinal value as the output FrameStore/Channel...
264  mAudioSystem = ::NTV2ChannelToAudioSystem(mConfig.fOutputChannel);
265 
266  if (mConfig.fNumAudioLinks > 1) // For users that want to send 32 or 64 audio channels on 2 or 4 SDI links
267  switch (mAudioSystem)
268  {
269  default:
270  case NTV2_AUDIOSYSTEM_1:
271  { const UWord numChan(NTV2_IS_QUAD_QUAD_HFR_VIDEO_FORMAT(mConfig.fVideoFormat) ? 4 : 2);
272  const NTV2AudioSystemSet audSystems (::NTV2MakeAudioSystemSet (mAudioSystem, numChan));
273  for (UWord chan(0); chan < numChan; chan++)
274  mDevice.SetSDIOutputAudioSystem (NTV2Channel(chan), NTV2AudioSystem(chan));
275  mDevice.SetNumberAudioChannels (numAudioChannels, audSystems);
276  mDevice.SetAudioBufferSize (NTV2_AUDIO_BUFFER_BIG, audSystems);
277  mDevice.SetAudioLoopBack (NTV2_AUDIO_LOOPBACK_OFF, audSystems);
278  break;
279  }
280  case NTV2_AUDIOSYSTEM_3:
281  mDevice.SetSDIOutputAudioSystem (NTV2_CHANNEL3, NTV2_AUDIOSYSTEM_3);
282  mDevice.SetSDIOutputAudioSystem (NTV2_CHANNEL4, NTV2_AUDIOSYSTEM_4);
283  break;
284  }
285  else
286  {
287  mDevice.SetSDIOutputAudioSystem (::NTV2MakeChannelSet (NTV2_CHANNEL1, 4), mAudioSystem);
288  mDevice.SetNumberAudioChannels (numAudioChannels, mAudioSystem);
289  mDevice.SetAudioBufferSize (NTV2_AUDIO_BUFFER_BIG, mAudioSystem);
290  mDevice.SetAudioLoopBack (NTV2_AUDIO_LOOPBACK_OFF, mAudioSystem);
291  }
292 
293  if (mConfig.fDoHDMIOutput)
294  {
295  mDevice.SetHDMIOutAudioRate(NTV2_AUDIO_48K);
296  mDevice.SetHDMIOutAudioFormat(NTV2_AUDIO_FORMAT_LPCM);
297  mDevice.SetHDMIOutAudioSource8Channel(NTV2_AudioChannel1_8, mAudioSystem);
298  }
299 
300  return AJA_STATUS_SUCCESS;
301 
302 } // SetUpAudio

305 AJAStatus NTV2Player8K::SetUpHostBuffers (void)
306 {
307  CNTV2DemoCommon::SetDefaultPageSize(); // Set host-specific page size
308 
309  // Let my circular buffer know when it's time to quit...
310  mFrameDataRing.SetAbortFlag (&mGlobalQuit);
311 
312  // Multi-link audio uses stacked buffers for transferring to the board,
313  // the first byte after the end of the first audio link buffer is the start of the second audio link buffer.
314  const size_t audioBufferSize (gAudMaxSizeBytes * uint32_t(mConfig.fNumAudioLinks));
315 
316  // Allocate and add each in-host NTV2FrameData to my circular buffer member variable...
317  mHostBuffers.reserve(CIRCULAR_BUFFER_SIZE);
318  while (mHostBuffers.size() < CIRCULAR_BUFFER_SIZE)
319  {
320  mHostBuffers.push_back(NTV2FrameData()); // Make a new NTV2FrameData...
321  NTV2FrameData & frameData(mHostBuffers.back()); // ...and get a reference to it
322 
323  // Don't allocate a page-aligned video buffer here.
324  // Instead, the test pattern buffers are used (and re-used) in the consumer thread.
325  // This saves a LOT of memory and time spent copying data with these large 4K/UHD rasters.
326  // NOTE: This differs substantially from the NTV2Player demo, which pre-allocates the ring of video buffers
327  // here, then in its producer thread, copies a fresh, unmodified test pattern raster into the video
328  // buffer, blits timecode into it, then transfers it to the hardware in its consumer thread.
329 
330  // Allocate a page-aligned audio buffer (if transmitting audio)
331  if (mConfig.WithAudio())
332  if (!frameData.fAudioBuffer.Allocate (audioBufferSize, BUFFER_PAGE_ALIGNED))
333  {
334  PLFAIL("Failed to allocate " << xHEX0N(audioBufferSize,8) << "-byte audio buffer");
335  return AJA_STATUS_MEMORY;
336  }
337  if (frameData.fAudioBuffer)
338  {
339  frameData.fAudioBuffer.Fill(ULWord(0));
340  #ifdef NTV2_BUFFER_LOCKING
341  mDevice.DMABufferLock(frameData.fAudioBuffer, /*alsoPreLockSGL*/true);
342  #endif
343  }
344  mFrameDataRing.Add (&frameData);
345  } // for each NTV2FrameData
346  return AJA_STATUS_SUCCESS;
347 
348 } // SetUpHostBuffers

630 void NTV2Player8K::ConsumeFrames (void)
631 {
632  ULWord acOptions (AUTOCIRCULATE_WITH_RP188);
633  AUTOCIRCULATE_TRANSFER outputXfer;
634  AUTOCIRCULATE_STATUS outputStatus;
635  AJAAncillaryData * pPkt (AJA_NULL);
636  ULWord goodXfers(0), badXfers(0), prodWaits(0), noRoomWaits(0);
637  const UWord numACFramesPerChannel(7);
638 
639  // Stop AutoCirculate, just in case someone else left it running...
640  mDevice.AutoCirculateStop(mConfig.fOutputChannel);
641  mDevice.WaitForOutputVerticalInterrupt(mConfig.fOutputChannel, 4); // Let it stop
642  PLNOTE("Thread started");
643 
644  if (pPkt)
645  { // Allocate page-aligned host Anc buffer...
646  uint32_t hdrPktSize (0);
647  if (!outputXfer.acANCBuffer.Allocate(gAncMaxSizeBytes, BUFFER_PAGE_ALIGNED) || !outputXfer.acANCBuffer.Fill(0LL))
648  PLWARN("Anc buffer " << xHEX0N(gAncMaxSizeBytes,8) << "(" << DEC(gAncMaxSizeBytes) << ")-byte allocate failed -- HDR anc insertion disabled");
649  else if (AJA_FAILURE(pPkt->GenerateTransmitData (outputXfer.acANCBuffer, outputXfer.acANCBuffer, hdrPktSize)))
650  {
651  PLWARN("HDR anc insertion disabled -- GenerateTransmitData failed");
652  outputXfer.acANCBuffer.Deallocate();
653  }
654  else
655  acOptions |= AUTOCIRCULATE_WITH_ANC;
656  }
657 #ifdef NTV2_BUFFER_LOCKING
658  if (outputXfer.acANCBuffer)
659  mDevice.DMABufferLock(outputXfer.acANCBuffer, /*alsoLockSGL*/true);
660 #endif
661 
662  // Calculate start & end frame numbers...
663  const UWord startNum (mConfig.fOutputChannel < 2 ? 0 : numACFramesPerChannel); // Ch1: frames 0-6
664  const UWord endNum (mConfig.fOutputChannel < 2 ? numACFramesPerChannel-1 : numACFramesPerChannel*2-1); // Ch5: frames 7-13
665  if (mConfig.fNumAudioLinks > 1)
666  {
667  acOptions |= AUTOCIRCULATE_WITH_MULTILINK_AUDIO1;
668  if (NTV2_IS_QUAD_QUAD_HFR_VIDEO_FORMAT(mConfig.fVideoFormat))
669  {
670  acOptions |= AUTOCIRCULATE_WITH_MULTILINK_AUDIO2;
671  acOptions |= AUTOCIRCULATE_WITH_MULTILINK_AUDIO3;
672  }
673  }
674 
675  // Initialize & start AutoCirculate...
676  bool initOK (mDevice.AutoCirculateInitForOutput (mConfig.fOutputChannel, 0, mAudioSystem, acOptions,
677  1 /*numChannels*/, startNum, endNum));
678  if (!initOK)
679  {PLFAIL("AutoCirculateInitForOutput failed"); mGlobalQuit = true;}
680 
681  while (!mGlobalQuit)
682  {
683  mDevice.AutoCirculateGetStatus (mConfig.fOutputChannel, outputStatus);
684 
685  // Check if there's room for another frame on the card...
686  if (outputStatus.CanAcceptMoreOutputFrames())
687  {
688  // Device has at least one free frame buffer that can be filled.
689  // Wait for the next frame in our ring to become ready to "consume"...
690  NTV2FrameData * pFrameData (mFrameDataRing.StartConsumeNextBuffer());
691  if (!pFrameData)
692  {prodWaits++; continue;}
693 
694  // Unlike in the NTV2Player demo, I now burn the current timecode into the test pattern buffer that was noted
695  // earlier into this FrameData in my Producer thread. This is done to avoid copying large 8K/UHD2 rasters.
696  const NTV2FrameRate ntv2FrameRate (::GetNTV2FrameRateFromVideoFormat(mConfig.fVideoFormat));
697  const TimecodeFormat tcFormat (CNTV2DemoCommon::NTV2FrameRate2TimecodeFormat(ntv2FrameRate));
698  const CRP188 rp188Info (mCurrentFrame++, 0, 0, 10, tcFormat);
699  NTV2_RP188 tcData;
700  string timeCodeString;
701 
702  rp188Info.GetRP188Reg (tcData);
703  rp188Info.GetRP188Str (timeCodeString);
704  mTCBurner.BurnTimeCode (pFrameData->fVideoBuffer, timeCodeString.c_str(), 80);
705 
706  // Transfer the timecode-burned frame (plus audio) to the device for playout...
707  outputXfer.acVideoBuffer.Set (pFrameData->fVideoBuffer, pFrameData->fVideoBuffer);
708  outputXfer.acAudioBuffer.Set (pFrameData->fAudioBuffer, pFrameData->fNumAudioBytes);
709  outputXfer.SetOutputTimeCode (tcData, ::NTV2ChannelToTimecodeIndex(mConfig.fOutputChannel, /*LTC=*/false, /*F2=*/false));
710  outputXfer.SetOutputTimeCode (tcData, ::NTV2ChannelToTimecodeIndex(mConfig.fOutputChannel, /*LTC=*/true, /*F2=*/false));
711 
712  // Perform the DMA transfer to the device...
713  if (mDevice.AutoCirculateTransfer (mConfig.fOutputChannel, outputXfer))
714  goodXfers++;
715  else
716  badXfers++;
717 
718  if (goodXfers == 3)
719  mDevice.AutoCirculateStart(mConfig.fOutputChannel);
720 
721  // Signal that the frame has been "consumed"...
722  mFrameDataRing.EndConsumeNextBuffer();
723  continue; // Back to top of while loop
724  }
725 
726  // Wait for one or more buffers to become available on the device, which should occur at next VBI...
727  noRoomWaits++;
728  mDevice.WaitForOutputVerticalInterrupt(mConfig.fOutputChannel);
729  } // loop til quit signaled
730 
731  // Stop AutoCirculate...
732  mDevice.AutoCirculateStop(mConfig.fOutputChannel);
733  PLNOTE("Thread completed: " << DEC(goodXfers) << " xfers, " << DEC(badXfers) << " failed, "
734  << DEC(prodWaits) << " starves, " << DEC(noRoomWaits) << " VBI waits");
735 
736 } // ConsumeFrames

763 void NTV2Player8K::ProduceFrames (void)
764 {
765  ULWord freqNdx(0), testPatNdx(0), badTally(0);
766  double timeOfLastSwitch (0.0);
767 
768  const AJATimeBase timeBase (CNTV2DemoCommon::GetAJAFrameRate(::GetNTV2FrameRateFromVideoFormat(mConfig.fVideoFormat)));
769  const NTV2TestPatternNames tpNames (NTV2TestPatternGen::getTestPatternNames());
770 
771  PLNOTE("Thread started");
772  while (!mGlobalQuit)
773  {
774  NTV2FrameData * pFrameData (mFrameDataRing.StartProduceNextBuffer());
775  if (!pFrameData)
776  { badTally++; // No frame available!
777  AJATime::Sleep(10); // Wait a bit for the consumer thread to free one up for me...
778  continue; // ...then try again
779  }
780 
781  // Unlike NTV2Player::ProduceFrames, NTV2Player8K::ProduceFrames doesn't touch this frame's video buffer.
782  // Instead, to avoid wasting time copying large 8K/UHD2 rasters, in this thread we simply note which test
783  // pattern buffer is to be modified and subsequently transferred to the hardware. This happens later, in
784  // NTV2Player8K::ConsumeFrames...
785  NTV2Buffer & testPatVidBuffer(mTestPatRasters.at(testPatNdx));
786  pFrameData->fVideoBuffer.Set(testPatVidBuffer.GetHostPointer(), testPatVidBuffer.GetByteCount());
787 
788  // If also playing audio...
789  if (pFrameData->AudioBuffer()) // ...then generate audio tone data for this frame...
790  pFrameData->fNumAudioBytes = AddTone(pFrameData->fAudioBuffer); // ...and remember number of audio bytes to xfer
791 
792  // Every few seconds, change the test pattern and tone frequency...
793  const double currentTime (timeBase.FramesToSeconds(mCurrentFrame++));
794  if (currentTime > timeOfLastSwitch + 4.0)
795  {
796  freqNdx = (freqNdx + 1) % gNumFrequencies;
797  testPatNdx = (testPatNdx + 1) % ULWord(mTestPatRasters.size());
798  mToneFrequency = gFrequencies[freqNdx];
799  timeOfLastSwitch = currentTime;
800  PLINFO("F" << DEC0N(mCurrentFrame,6) << ": tone=" << mToneFrequency << "Hz, pattern='" << tpNames.at(testPatNdx) << "'");
801  } // if time to switch test pattern & tone frequency
802 
803  // Signal that I'm done producing this FrameData, making it immediately available for transfer/playout...
804  mFrameDataRing.EndProduceNextBuffer();
805 
806  } // loop til mGlobalQuit goes true
807  PLNOTE("Thread completed: " << DEC(mCurrentFrame) << " frame(s) produced, " << DEC(badTally) << " failed");
808 
809 } // ProduceFrames

813 {

838  mToneFrequency, // tone frequency [Hz]
839  31, // bits per sample
840  false, // don't byte swap
