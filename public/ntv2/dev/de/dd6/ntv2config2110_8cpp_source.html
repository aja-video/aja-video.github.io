514

517  NTV2Standard std = fd.GetVideoStandard();

520  uint32_t val = ( (((uint32_t) fr) << 8) |

973  ppp = 0x0130;
974  }

1001  }
1002  else if (StreamType(stream) == AUDIO_STREAM)
