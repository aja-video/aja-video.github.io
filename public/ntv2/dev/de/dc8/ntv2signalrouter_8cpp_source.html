36 static NTV2StringList & Tokenize (const string & inString, NTV2StringList & outTokens, const string & inDelimiters = " ", bool inTrimEmpty = false)
37 {
38  string::size_type pos (0), lastPos (0);
39  outTokens.clear ();
40  while (true)
41  {
42  pos = inString.find_first_of (inDelimiters, lastPos);
43  if (pos == string::npos)
44  {
45  pos = inString.length ();
46  if (pos != lastPos || !inTrimEmpty)
47  outTokens.push_back (NTV2StringList::value_type (inString.data () + lastPos, NTV2StringList::size_type(pos - lastPos)));
48  break;
49  }
50  else
51  {
52  if (pos != lastPos || !inTrimEmpty)
53  outTokens.push_back (NTV2StringList::value_type (inString.data () + lastPos, NTV2StringList::size_type(pos - lastPos)));
54  }
55  lastPos = pos + 1;
56  }
57  return outTokens;
58 }

108 bool CNTV2SignalRouter::ResetFromRegisters (const NTV2InputXptIDSet & inInputs, const NTV2RegisterReads & inRegReads)
109 {
110  Reset();
111  for (NTV2InputXptIDSetConstIter it(inInputs.begin()); it != inInputs.end(); ++it)
112  {
113  uint32_t regNum(0), maskNdx(0);
114  CNTV2RegisterExpert::GetCrosspointSelectGroupRegisterInfo (*it, regNum, maskNdx);
115  NTV2RegisterReadsConstIter iter (::FindFirstMatchingRegisterNumber(regNum, inRegReads));
116  if (iter == inRegReads.end())
117  continue;
118 
119  NTV2_ASSERT(iter->registerNumber == regNum);
120  NTV2_ASSERT(iter->registerMask == 0xFFFFFFFF);
121  NTV2_ASSERT(iter->registerShift == 0);
122  NTV2_ASSERT(maskNdx < 4);
123  const uint32_t regValue (iter->registerValue & sSignalRouterRegMasks[maskNdx]);
124  const NTV2OutputXptID outputXpt (NTV2OutputXptID(regValue >> sSignalRouterRegShifts[maskNdx]));
125  if (outputXpt != NTV2_XptBlack)
126  mConnections.insert(NTV2SignalConnection (*it, outputXpt));
127  } // for each NTV2InputXptID
128  return true;

164 bool CNTV2SignalRouter::Compare (const CNTV2SignalRouter & inRHS, NTV2XptConnections & outNew,
165  NTV2XptConnections & outChanged, NTV2XptConnections & outMissing) const
166 {
167  outNew.clear(); outChanged.clear(); outMissing.clear();
168  // Check that my connections are also in RHS:
169  for (NTV2XptConnectionsConstIter it(mConnections.begin()); it != mConnections.end(); ++it)
170  {
171  const NTV2SignalConnection & connection (*it);
172  const NTV2InputXptID inputXpt(connection.first);
173  const NTV2OutputXptID outputXpt(connection.second);
174  if (inRHS.HasConnection(inputXpt, outputXpt))
175  ;
176  else if (inRHS.HasInput(inputXpt))
177  outChanged.insert(NTV2Connection(inputXpt, inRHS.GetConnectedOutput(inputXpt))); // Connection changed from this
178  else
179  outNew.insert(connection); // Connection is new in me, not in RHS
180  }
181 
182  // Check that RHS' connections are also in me...
183  const NTV2XptConnections connectionsRHS(inRHS.GetConnections());
184  for (NTV2XptConnectionsConstIter it(connectionsRHS.begin()); it != connectionsRHS.end(); ++it)
185  {
186  const NTV2SignalConnection & connectionRHS (*it);
187  const NTV2InputXptID inputXpt(connectionRHS.first);
188  const NTV2OutputXptID outputXpt(connectionRHS.second);
189  NTV2XptConnectionsConstIter pFind (mConnections.find(inputXpt));
190  if (pFind == mConnections.end()) // If not found in me...
191  outMissing.insert(connectionRHS); // ...in RHS, but missing in me
192  else if (pFind->second != outputXpt) // If output xpt differs...
193  outChanged.insert(connectionRHS); // ...then 'connection' is changed (in RHS, not in me)
194  }
195 
196  return outNew.empty() && outChanged.empty() && outMissing.empty(); // Return true if identical
197 }

601 bool CNTV2SignalRouter::GetConnectionsFromRegs (const NTV2InputXptIDSet & inInputXptIDs, const NTV2RegisterReads & inRegValues, NTV2XptConnections & outConnections)
602 {
603  outConnections.clear();
604  for (NTV2InputXptIDSetConstIter it(inInputXptIDs.begin()); it != inInputXptIDs.end(); ++it)
605  {
606  uint32_t regNum(0), maskNdx(0);
607  CNTV2RegisterExpert::GetCrosspointSelectGroupRegisterInfo (*it, regNum, maskNdx);
608  NTV2RegisterReadsConstIter iter (::FindFirstMatchingRegisterNumber(regNum, inRegValues));
609  if (iter == inRegValues.end())
610  continue;
611 
612  if (iter->registerNumber != regNum)
613  return false; // Register numbers must match here
614  if (iter->registerMask != 0xFFFFFFFF)
615  return false; // Mask must be 0xFFFFFFFF
616  if (iter->registerShift)
617  return false; // Shift must be zero
618  NTV2_ASSERT(maskNdx < 4);
619  const uint32_t regValue (iter->registerValue & sSignalRouterRegMasks[maskNdx]);
620  const NTV2OutputXptID outputXpt (NTV2OutputXptID(regValue >> sSignalRouterRegShifts[maskNdx]));
621  if (outputXpt != NTV2_XptBlack)
622  outConnections.insert(NTV2SignalConnection (*it, outputXpt));
623  } // for each NTV2InputXptID
624  return true;
625 }

628 bool CNTV2SignalRouter::CompareConnections (const NTV2XptConnections & inLHS,
629  const NTV2XptConnections & inRHS,
630  NTV2XptConnections & outNew,
631  NTV2XptConnections & outMissing)
632 {
633  outNew.clear(); outMissing.clear();
634  // Check that LHS connections are also in RHS:
635  for (NTV2XptConnectionsConstIter it(inLHS.begin()); it != inLHS.end(); ++it)
636  {
637  const NTV2SignalConnection & LHSconnection(*it);
638  const NTV2InputXptID inputXpt(LHSconnection.first);
639  const NTV2OutputXptID outputXpt(LHSconnection.second);
640  NTV2XptConnectionsConstIter RHSit(inRHS.find(inputXpt));
641  if (RHSit == inRHS.end())
642  outMissing.insert(LHSconnection); // LHSConnection's inputXpt missing from RHS
643  else if (RHSit->second == outputXpt)
644  ; // LHS's input xpt connected to same output xpt as RHS
645  else
646  {
647  outMissing.insert(LHSconnection); // LHS connection missing from RHS
648  outNew.insert(*RHSit); // RHS connection is new
649  }
650  }
651 
652  // Check that RHS connections are also in LHS...
653  for (NTV2XptConnectionsConstIter it(inRHS.begin()); it != inRHS.end(); ++it)
654  {
655  const NTV2SignalConnection & connectionRHS (*it);
656  const NTV2InputXptID inputXpt(connectionRHS.first);
657  const NTV2OutputXptID outputXpt(connectionRHS.second);
658  NTV2XptConnectionsConstIter LHSit(inLHS.find(inputXpt));
659  if (LHSit == inLHS.end()) // If RHS input xpt not in LHS...
660  outNew.insert(connectionRHS); // ...then RHS connection is new
661  else if (LHSit->second != outputXpt) // Else if output xpt changed...
662  // Should've already been handled in previous for loop
663  NTV2_ASSERT(outMissing.find(LHSit->first) != outMissing.end() && outNew.find(LHSit->first) != outNew.end());
664  }
665 
666  return outNew.empty() && outMissing.empty(); // Return true if identical
667 }

670 bool CNTV2SignalRouter::CreateFromString (const string & inString, NTV2XptConnections & outConnections) // STATIC
671 {
672  NTV2StringList lines;
673  string stringToParse(inString); aja::strip(aja::lower(stringToParse));
674  aja::replace(stringToParse, " ", "");
675  aja::replace(stringToParse, "\t", "");
676  aja::replace(stringToParse, "&lt;","<"); // in case uuencoded
677 
678  outConnections.clear();
679  if (Tokenize(stringToParse, lines, "\n\r", true).empty()) // Split the string at line breaks
680  {
681  SRWARN("No lines resulted from input string '" << stringToParse << "'");
682  return true; // Nothing there
683  }
684 
685  if (lines.front().find("<==") != string::npos)
686  {
687 // SRDBG(lines.size() << " lines");
688  for (NTV2StringListConstIter pEachLine(lines.begin()); pEachLine != lines.end(); ++pEachLine)
689  {
690 // SRDBG(" line '" << *pEachLine << "'");
691  size_t pos (pEachLine->find("<=="));
692  if (pos == string::npos)
693  {SRFAIL("Parse error: '<==' missing in line '" << *pEachLine << "'"); return false;}
694  string leftPiece (pEachLine->substr(0, pos)); aja::strip(leftPiece);
695  string rightPiece (pEachLine->substr(pos + 3, pEachLine->length())); aja::strip(rightPiece);
696  NTV2InputXptID inputXpt (StringToNTV2InputCrosspointID(leftPiece));
697  NTV2OutputXptID outputXpt (StringToNTV2OutputCrosspointID(rightPiece));
698  //SRDBG(" L'" << leftPiece << "', R'" << rightPiece << "'");
699  if (inputXpt == NTV2_INPUT_CROSSPOINT_INVALID)
700  {SRFAIL("Parse error: invalid input crosspoint from '" << leftPiece << "' from line '" << *pEachLine << "'"); return false;}
701  if (outConnections.find(inputXpt) != outConnections.end())
702  SRWARN("Overwriting " << ::NTV2InputCrosspointIDToString(inputXpt) << "-" << ::NTV2OutputCrosspointIDToString(outConnections[inputXpt])
703  << " with " << ::NTV2InputCrosspointIDToString(inputXpt) << "-" << ::NTV2OutputCrosspointIDToString(outputXpt));
704  outConnections.insert(NTV2Connection(inputXpt, outputXpt));
705  } // for each line
706  }
707  else if (lines.front().find("connect(") != string::npos)
708  {
709  for (NTV2StringListConstIter pLine(lines.begin()); pLine != lines.end(); ++pLine)
710  {
711  string line(*pLine); aja::strip(line);
712  if (line.empty())
713  continue;
714 // SRDBG(" line '" << line << "'");
715  size_t openParenPos(line.find("(")), closedParenPos(line.find(");"));
716  if (openParenPos == string::npos || closedParenPos == string::npos || openParenPos > closedParenPos)
717  {SRFAIL("Parse error: '(' or ');' missing in line '" << line << "'"); return false;}
718  string remainder(line.substr(openParenPos+1, closedParenPos - openParenPos - 1));
719  NTV2StringList xptNames;
720  aja::split(remainder, ',', xptNames);
721  if (xptNames.size() < 2 || xptNames.size() > 2)
722  {SRFAIL("Parse error: " << DEC(xptNames.size()) << " 'Connect' parameter(s) found, expected 2"); return false;}
723  NTV2InputXptID inputXpt (StringToNTV2InputCrosspointID(xptNames.at(0)));
724  NTV2OutputXptID outputXpt (StringToNTV2OutputCrosspointID(xptNames.at(1)));
725  //SRDBG(" L'" << xptNames.at(0) << "', R'" << xptNames.at(1) << "'");
726  if (inputXpt == NTV2_INPUT_CROSSPOINT_INVALID)
727  {SRFAIL("Parse error: invalid input crosspoint from '" << xptNames.at(0) << "' from line '" << *pLine << "'"); return false;}
728  if (outputXpt == NTV2_OUTPUT_CROSSPOINT_INVALID)
729  {SRFAIL("Parse error: invalid output crosspoint from '" << xptNames.at(1) << "' from line '" << *pLine << "'"); return false;}
730  if (outConnections.find(inputXpt) != outConnections.end())
731  SRWARN("Overwriting " << ::NTV2InputCrosspointIDToString(inputXpt) << "-" << ::NTV2OutputCrosspointIDToString(outConnections[inputXpt])
732  << " with " << ::NTV2InputCrosspointIDToString(inputXpt) << "-" << ::NTV2OutputCrosspointIDToString(outputXpt));
733  outConnections.insert(NTV2Connection(inputXpt, outputXpt));
734  } // for each line
735  }
736  else
737  {SRFAIL("Unable to parse '" << lines.front() << "' -- expected '.contains(' or '<=='"); return false;}
738  SRINFO(DEC(outConnections.size()) << " connection(s) created from input string");
739  return true;
740 }

1026  NTV2InputXptID & outInputXpt, NTV2OutputXptIDSet & outOutputXpts, const bool inAppendOutputXpts)
