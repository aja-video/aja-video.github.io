174 uint32_t AJA_CalcRowBytesForFormat(AJA_PixelFormat format, uint32_t width)
175 {
176  int rowBytes = 0;
177 
178  switch (format)
179  {
180  case AJA_PixelFormat_YCbCr8:
181  case AJA_PixelFormat_YUY28:
182  rowBytes = width * 2;
183  break;
184 
185  case AJA_PixelFormat_YCbCr10:
186  case AJA_PixelFormat_YCbCr_DPX:
187  rowBytes = (( width % 48 == 0 ) ? width : (((width / 48 ) + 1) * 48)) * 8 / 3;
188  break;
189 
190  case AJA_PixelFormat_RGB8_PACK:
191  case AJA_PixelFormat_BGR8_PACK:
192  rowBytes = width * 3;
193  break;
194 
195  case AJA_PixelFormat_ARGB8:
196  case AJA_PixelFormat_RGBA8:
197  case AJA_PixelFormat_ABGR8:
198  case AJA_PixelFormat_RGB10:
199  case AJA_PixelFormat_RGB10_PACK:
200  case AJA_PixelFormat_RGB_DPX:
201  case AJA_PixelFormat_RGB_DPX_LE:
202  rowBytes = width * 4;
203  break;
204 
205  case AJA_PixelFormat_RGB16:
206  // Note: not technically true but for now using this for bayer test patterns.
207  case AJA_PixelFormat_BAYER10_DPX_LJ:
208  case AJA_PixelFormat_BAYER12_DPX_LJ:
209  case AJA_PixelFormat_BAYER10_HS:
210  case AJA_PixelFormat_BAYER12_HS:
211  rowBytes = width * 6;
212  break;
213 
214  case AJA_PixelFormat_RGB12:
215  // 36 bits/pixel packed.
216  rowBytes = width * 36 / 8;
217  break;
218 
219  case AJA_PixelFormat_RAW10:
220  case AJA_PixelFormat_RAW10_HS:
221  // 10 bits/pixel raw packed

223  break;
